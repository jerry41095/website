Dependency

type Dependency {
  name: String
  scope: DependencyScope
  versionRange: String
}

Fields

name (String)

Name of the package specified by the import statement

scope (DependencyScope)

“BUILD” or “TEST” corresponding to Imports vs. TestImports

versionRange (String)

Reserved for future use

DependencyConnection

type DependencyConnection {
  edges: [DependencyEdge]
  pageInfo: PageInfo!
}

A connection to a list of dependencies.

Fields

edges ([DependencyEdge])

Edges connecting to the listed Dependency nodes

pageInfo (PageInfo!)

Pagination metadata for this connection

Package

type Package {
  name: String
  version(version: String, revision: String): Version
  versions(first: Int, last: Int, before: String, after: String): VersionConnection
}

Fields

name (String)

Package name as specified in the import statement

version (Version)

Query for a specific version by either the Semantic Version number, or by the revision ID of the repository.

Argument Type Description
revision String Find by repo revision
version String Find by version number

versions (VersionConnection)

List all tagged and latest auto-generated versions of this package.

Argument Type Description
before String Cursor for pagination
after String Cursor for pagination
first Int Page size
last Int Page size

Cursor for last item of this page

Release

type Release {
  name: String
  version: String
  url: String
}

Fields

name (String)

Name of the repository from which the package was extracted

version (String)

Convenience accessor to the parent Version version

url (String)

Link to download an archive of this release in tgz form

Version

type Version {
  name: String
  version: String
  release: Release
  dependencies(after: String, first: Int, last: Int, before: String): DependencyConnection
}

Fields

name (String)

Convenience accessor to the parent Package name

version (String)

Semantic version derrived from repository tags and history

release (Release)

Details about the archive needed to install this package

dependencies (DependencyConnection)

A RelayConnection to a list of dependencies that are imported by this version.

Argument Type Description
before String Cursor for pagination
after String Cursor for pagination
first Int Page size
last Int Page size
